#!/usr/bin/env ruby -w
# encoding: UTF-8
描述 '单元测试' do
  
  之前 {@a = 5}
  
  之后 {@a = nil}
  
  描述 "数字" do
    设置(:数字) {5}
    
    它 "应该是5" do
      期望(数字).等于 5
    end
  end
  
  添加测试 "断言测试" do
    期望([]).为空
    期望([1]).不为空
    期望(1).等于 1
    期望(1).不等于 2
    期望(1).接近 1
    期望(1).不接近 2
    期望([1]).包含 1
    期望([1]).不包含 2
    期望(1).为实例 Fixnum
    期望(1).非实例 Integer
    期望(1).属于 Integer
    期望(1).不属于 String
    期望('1').匹配 /\d/
    期望('1').不匹配 /a/
    期望(nil).为无
    期望('').不为无
    期望(1).应 :<, 2
    期望(1).应不 :>, 1
    期望(proc {print 'abc'}).应输出 'abc'
    期望(proc {}).不输出
    期望(proc {引发 E异常}).应引发 E异常
    期望(1).响应 :+
    期望(1).不响应 :hahaha
    期望(:a).相同于 :a
    期望('a').不同于 'a'
    期望(proc {抛出 :test}).应抛出 :test
  end
end

R单元测试.运行#(['-v'])
