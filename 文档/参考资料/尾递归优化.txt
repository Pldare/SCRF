RubyVM::InstructionSequence.compile_option = {
  :tailcall_optimization => true,
  :trace_instruction => false
}
code = <<EOF
  $i = 1
  def a
    do_sth
    a
  end
  
  def do_sth
    if ($i == 10000)
      def a
        'modifying a'
      end
    else
      $i += 1
    end
  end
  a
EOF
t = Time.now
1000.times{ RubyVM::InstructionSequence.compile(code).eval }
p Time.now - t
